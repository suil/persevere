# Key insights
* Transform subset problem into target sum using half of total array sum
* Apply recursive backtracking with include/exclude decision for each element
* Use memoization to cache subproblem results and avoid redundant calculations
* Early termination when target becomes 0 (found) or negative (impossible)

# Code
```javascript
var canPartition = function(nums) {
    const sum = nums.reduce((sum, n) => sum + n, 0);
    if (sum % 2 !== 0) {
        return false;
    }
    return memoization(nums, Math.floor(sum / 2), new Map());
};

function memoization(nums, target, memo) {
    if (memo.has(`${nums[0]}-${target}`)) {
        return memo.get(`${nums[0]}-${target}`);
    }
    if (target === 0) {
        return true;
    }
    if (nums.length === 0 || target < 0) {
        return false;
    }
    
    const taken = memoization(nums.slice(1), target - nums[0], memo);
    const notTaken = memoization(nums.slice(1), target, memo);
    const res = taken || notTaken;
    memo.set(`${nums[0]}-${target}`, res);
    return res;
}
```
```javascript
var canPartition = function(nums) {
    const sum = nums.reduce((memo, num) => memo + num, 0);
    const subsetSum = Math.floor(sum / 2);
    if (sum % 2 != 0) { return false; }
    
    const dp = [...Array(subsetSum + 1)].fill(false);
    dp[0] = true;
    
    for (const num of nums) {
        for (let i = subsetSum; i >= num; i--) {
            dp[i] = dp[i - num] || dp[i];
        }
    }
    
    return dp[subsetSum];
};
```

# Complexity
**Solution 1:**
**Time:** O(n * sum) - explores each combination of array index and target value once due to memoization
**Space:** O(n * sum) - stores memoization results plus O(nÂ²) for recursive call stack and array slicing

**Solution 2:**
**Time:** O(n * sum) - iterates through each number and each possible sum value
**Space:** O(sum) - uses a 1D DP array of size equal to target sum
