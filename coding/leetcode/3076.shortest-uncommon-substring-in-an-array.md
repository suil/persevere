# Shortest Uncommon Substring in an Array

You are given an array arr of size n consisting of non-empty strings.

/**
 * @param {string[]} arr
 * @return {string[]}
 */
var shortestSubstrings = function(arr) {
    
    const map = new Map();
    const arrWithSubstrings = [];
    
    for (let i = 0; i < arr.length; i++) {
        const substrs = new Set();
        const str = arr[i];
        for (let start = 0; start < str.length; start++) {
            for (let end = start; end < str.length; end++) {
                const substr = str.substring(start, end + 1);
                substrs.add(substr);
                if (!map.has(substr)) map.set(substr, new Set());
                map.get(substr).add(i);
            }
        }
        arrWithSubstrings[i] = [...substrs].sort((a, b) => a.length - b.length || a.localeCompare(b));
    }

    const res = Array(arr.length).fill('');
    for (let i = 0; i < arrWithSubstrings.length; i++) {
        for (const substr of arrWithSubstrings[i]) {
            if (map.get(substr).size === 1 && map.get(substr).has(i)) { // from itself
                res[i] = substr;
                break;
            }
        }
    }

    return res;
};
