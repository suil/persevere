```javascript
/**
 * @param {number[]} g
 * @param {number[]} s
 * @return {number}
 */
var findContentChildren = function(g, s) {
    g.sort((a, b) => b - a);
    s.sort((a, b) => b - a);

    let count = 0;
    let si = 0;
    for (let gi = 0; gi < g.length; gi++) {
        if (s[si] >= g[gi]) {
            count++;
            si++;
        }
    }
    return count;
};
```
