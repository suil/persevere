# Subsequence of Size K With the Largest Even Sum
[2098. Subsequence of Size K With the Largest Even Sum](https://leetcode.com/problems/subsequence-of-size-k-with-the-largest-even-sum/)

You are given an integer array nums and an integer k. Find the largest even sum of any subsequence of nums that has a length of k.

# Code
```javascript
/**
 * @param {number[]} nums
 * @param {number} k
 * @return {number}
 */
var largestEvenSum = function(nums, k) {
    // Sort nums in descending order
    nums.sort((a, b) => b - a);

    // Take the first k elements as initial subsequence
    let subseq = nums.slice(0, k);
    let remaining = nums.slice(k);

    // Compute sum of the initial subsequence
    let sum = subseq.reduce((a, b) => a + b, 0);

    // If it's even, return it
    if (sum % 2 === 0) return sum;

    const reversedSubSeq = subseq.reverse();
    // Identify candidates for swap to make sum even
    const smallestOddInSub = reversedSubSeq.find(num => num % 2 === 1);
    const smallestEvenInSub = reversedSubSeq.find(num => num % 2 === 0);

    const largestEvenInRest = remaining.find(num => num % 2 === 0);
    const largestOddInRest = remaining.find(num => num % 2 === 1);

    let candidateSums = [];

    // Try swapping odd from subseq with even from rest
    if (smallestOddInSub !== undefined && largestEvenInRest !== undefined) {
        candidateSums.push(sum - smallestOddInSub + largestEvenInRest);
    }

    // Try swapping even from subseq with odd from rest
    if (smallestEvenInSub !== undefined && largestOddInRest !== undefined) {
        candidateSums.push(sum - smallestEvenInSub + largestOddInRest);
    }

    return candidateSums.length > 0 ? Math.max(...candidateSums) : -1;
};
```
