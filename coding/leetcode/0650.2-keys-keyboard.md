# Key insights
* Dynamic programming with memoization to avoid recomputing subproblems
* Recursive state space exploration with (clipboard_length, current_string_length)
* At each step choose between copy-all or paste operations to minimize total moves
* Pruning invalid states where clipboard/string length exceeds target

```javascript
function memoize(n, clipboardLen, strLen, memo) {
    if (strLen === n) { return 0; }

    if (clipboardLen > n || strLen > n) { return Infinity; }

    const copy = strLen > clipboardLen ? memoize(n, strLen, strLen, memo) : Infinity;
    const paste = clipboardLen > 0 ? dfs(n, clipboardLen, strLen + clipboardLen, memo) : Infinity;

    return Math.min(copy, paste) + 1;
}
```

```javascript
function minStepsBF(n) {
    if (n == 1) return 0;
    for (let i = 2; i <= Math.sqrt(n); i++) {
        if (n % i == 0) return i + minSteps(n / i);
    }
    return n;
}
```

```java
public int minSteps(int n) {
    int[] dp = new int[n + 1];
    int h = (int) Math.sqrt(n);
    for (int i = 2; i <= n; i++) {
        dp[i] = i;
        for (int j = 2; j <= h; j++) {
            if (i % j == 0) {
                dp[i] = dp[j] + dp[i / j];
                break;
            }
        }
    }
    return dp[n];
}
```
