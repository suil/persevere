# Odd String Difference
[2451. Odd String Difference](https://leetcode.com/problems/odd-string-difference/)

You are given an array of equal-length strings words. Assume that the length of each string is n.

/**
 * @param {string[]} words
 * @return {string}
 */
var oddString = function(words) {
    function getDiff(word) {
        const res = [];
        for (let i = 1; i < word.length; i++) {
            res.push(word[i].charCodeAt(0) - word[i - 1].charCodeAt(0));
        }
        return res.join('-');
    }
    
    const diff0 = getDiff(words[0]);
    const diff1 = getDiff(words[1]);
    const diff2 = getDiff(words[2]);
    let common = diff0 === diff1 ? diff0 : diff2 === diff0 ? diff0 : diff1;

    for (let i = 0; i < words.length; i++) {
        if (common !== getDiff(words[i])) {
            return words[i];
        }
    }
};
