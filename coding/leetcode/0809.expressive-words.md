# Key insights
- Two-pointer approach comparing character groups between the stretched string S and each word
- For each group of consecutive identical characters, the stretch is valid if S's group length >= word's group length AND S's group length >= 3 (or they are equal)
- Both pointers must reach the end of their strings for the word to be stretchy

# Code
```javascript
/**
 * @param {string} S
 * @param {string[]} words
 * @return {number}
 */
var expressiveWords = function(S, words) {
    return words.reduce((memo, word) => memo + (isStretchy(S, word) ? 1 : 0), 0);
};

function isStretchy(S, word) {
    let posS = 0;
    let posWord = 0;
    
    while (posS < S.length && posWord < word.length) {
        const charS = S[posS];
        const charWord = word[posWord];
        
        if (charS !== charWord) {
            return false;
        }
        
        let stretchLenS = 0;
        while (stretchLenS + posS < S.length && S[stretchLenS + posS] === charS) {
            stretchLenS++;
        }

        let stretchLenWord = 0;
        while (stretchLenWord + posWord < word.length && word[stretchLenWord + posWord] === charWord) {
            stretchLenWord++;
        }
        
        if (stretchLenS < stretchLenWord || (stretchLenS < 3 && stretchLenS !== stretchLenWord)) {
            return false;
        }
        
        posS += stretchLenS;
        posWord += stretchLenWord;
    }
    
    return posS === S.length && posWord === word.length;
}
```

# Complexity
**Time:** O(W * max(|S|, |word|)) where W is the number of words
**Space:** O(1)
