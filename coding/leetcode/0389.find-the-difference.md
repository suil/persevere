# Code
```javascript
/**
 * @param {string} s
 * @param {string} t
 * @return {character}
 */
var findTheDifference = function(s, t) {
    let tracker = new Array(26);
    tracker.fill(0);
    
    let aCode = 97;
    
    for (let i = 0; i<s.length; i++) {
        tracker[s.charCodeAt(i) - aCode] +=1
        tracker[t.charCodeAt(i) - aCode] -=1;
    }

    tracker[t.charCodeAt(t.length - 1) - aCode] -=1;
    
    return String.fromCharCode(tracker.indexOf(-1) + aCode);
};
```
