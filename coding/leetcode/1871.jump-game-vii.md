/**
 * @param {string} s
 * @param {number} minJump
 * @param {number} maxJump
 * @return {boolean}
 */
var canReach = function(s, minJump, maxJump) {
    const n = s.length;
    const queue = [0];
    let farthest = 0;

    while (queue.length > 0) {
        const i = queue.shift();
        let start = Math.max(i + minJump, farthest + 1);

        for (let j = start; j <= Math.min(i + maxJump, n - 1); j++) {
            if (s[j] === '0') {
                if (j === n - 1) return true;
                queue.push(j);
            }
        }

        farthest = i + maxJump;
    }

    return false;
};
