# Code
```javascript
/**
 * @param {number[]} nums
 * @return {number}
 */
var maxScore = function(nums) {
    if (nums.length < 2) return 0;

    let minTwoSize = Infinity;
    let minOneSize = nums.at(-1);
    let totalSum = nums.at(-1);

    for (let i = 0; i < nums.length - 1; i++) {
        const next2Sum = nums[i] + nums[i + 1];
        totalSum += nums[i];
        minTwoSize = Math.min(minTwoSize, next2Sum)
        minOneSize = Math.min(minOneSize, nums[i])
    }

    if (nums.length % 2 === 0) return totalSum - minTwoSize
    return totalSum - minOneSize
};
```
