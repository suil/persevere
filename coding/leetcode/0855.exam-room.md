/**
 * @param {number} n
 */
var ExamRoom = function(n) {
    this.n = n;
    this.occupied = [];
};

function getLowerBound(arr, target) {
    let left = 0;
    let right = arr.length - 1;
    while (left <= right) {
        const mid = Math.floor((left + right) / 2);
        if (arr[mid] < target) {
            left = mid + 1;
        } else {
            right = mid - 1;
        }
    }
    return left;
}
/**
 * @return {number}
 */
ExamRoom.prototype.seat = function() {
    if (this.occupied.length === 0) {
        this.occupied.push(0);
        return 0;
    }

    let maxDist = this.occupied[0];
    let maxIndex = 0;
    for (let i = 0; i < this.occupied.length - 1; i++) {
        const a = this.occupied[i];
        const b = this.occupied[i + 1];
        const dist = Math.floor((b - a) / 2);
        if (dist > maxDist) {
            maxDist = dist;
            maxIndex = a + dist;
        }
    }

    const last = this.occupied.at(-1);
    const lastDist = (this.n - 1) - last;
    if (lastDist > maxDist) {
        maxDist = lastDist;
        maxIndex = this.n - 1;
    }

    const index = getLowerBound(this.occupied, maxIndex);
    if (index === -1) {
        this.occupied.push(maxIndex);
    } else {
        this.occupied.splice(index, 0, maxIndex);
    }
    return maxIndex;
};

/** 
 * @param {number} p
 * @return {void}
 */
ExamRoom.prototype.leave = function(p) {
    let idx = this.occupied.indexOf(p);
    if (idx !== -1) this.occupied.splice(idx, 1);
};
