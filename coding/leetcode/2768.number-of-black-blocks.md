/**
 * @param {number} m
 * @param {number} n
 * @param {number[][]} coordinates
 * @return {number[]}
 */
var countBlackBlocks = function(m, n, coordinates) {
    
    const blockCount = new Map();

    const directions = [[-1, -1], [-1, 0], [0, -1], [0, 0]];

    for (const [blockRow, blockCol] of coordinates) {
        for (const [dRow, dCol] of directions) {
            const newRow = blockRow + dRow;
            const newCol = blockCol + dCol;
            if (newRow >= 0 && newRow < m - 1 && newCol >= 0 && newCol < n - 1) {
                blockCount.set(`${newRow}-${newCol}`, (blockCount.get(`${newRow}-${newCol}`) || 0) + 1);
            }
        }
    }

    res = Array(5).fill(0);
    for (const [key, count] of blockCount) {
        if (count <= 4) {
            res[count]++;
        }
    }
    res[0] = (m - 1) * (n - 1) - blockCount.size;
    return res;
};
