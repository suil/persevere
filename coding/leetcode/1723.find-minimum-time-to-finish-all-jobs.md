# Find Minimum Time to Finish All Jobs
[1723. Find Minimum Time to Finish All Jobs](https://leetcode.com/problems/find-minimum-time-to-finish-all-jobs/)

You are given an integer array jobs, where jobs[i] is the amount of time it takes to complete the ith job.

# Code
```javascript
/**
 * @param {number[]} jobs
 * @param {number} k
 * @return {number}
 */
var minimumTimeRequired = function(jobs, k) {
    jobs.sort((a, b) => b - a); // Assign big jobs first for pruning
    let left = jobs[0]; // Math.max(...jobs);
    let right = jobs.reduce((s, b) => s + b);

    function canFinish(limit) {
        function backtrack(index, loads) {
            if (index === jobs.length) return true;

            for (let i = 0; i < loads.length; i++) {
                if (loads[i] + jobs[index] <= limit) {
                    let newLoads = [...loads];
                    newLoads[i] += jobs[index];
                    if (backtrack(index + 1, newLoads)) return true;
                }

                if (loads[i] === 0) break;
            }

            return false;
        }
        return backtrack(0, new Array(k).fill(0));
    }

    let minMaxTime = right;
    while (left <= right) {
        let mid = Math.floor((left + right) / 2);
        if (canFinish(mid)) {
            minMaxTime = mid;
            right = mid - 1;
        } else {
            left = mid + 1;
        }
    }

    return minMaxTime;
};
```
