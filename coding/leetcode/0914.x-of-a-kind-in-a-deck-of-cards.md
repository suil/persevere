/**
 * @param {number[]} deck
 * @return {boolean}
 */
var hasGroupsSizeX = function(deck) {
    const freqMap = new Map();
    for (const card of deck) {
        freqMap.set(card, (freqMap.get(card) || 0) + 1);
    }
    
    let g;
    for (const [deck, count] of freqMap) {
        g = (g === undefined) ? count : gcd(g, count);
    }
    return g >= 2;
};

function gcd(x, y) {
    return x === 0 ? y : gcd(y % x, x)
}
