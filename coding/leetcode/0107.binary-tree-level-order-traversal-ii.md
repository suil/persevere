# Key insights
* Use BFS with queue to traverse tree level by level
* Process entire level at once using separate queue for next level
* Use unshift() to prepend levels for bottom-up order
* Level-by-level processing ensures correct reverse ordering

# Code
```javascript
var levelOrderBottom = function(root) {
    const output = [];
    let queue = [root];
    while (queue.length > 0) {
        const nextQueue = [];
        const level = [];
        for (const node of queue) {
            if (node === null) { continue; }
            level.push(node.val);
            nextQueue.push(node.left);
            nextQueue.push(node.right);
        }
        
        if (level.length > 0) { output.unshift(level); }
        queue = nextQueue;
    }
    return output;
};
```

```javascript
var levelOrderBottom = function(root) {
    const output = [];
    levelOrderBottomHelper(root, 0, output);
    return output.reverse();
};
function levelOrderBottomHelper(node, level, output) {
    if (node === null) { return; }
    if (!output[level]) { output[level] = []; }
    output[level].push(node.val);
    levelOrderBottomHelper(node.left, level + 1, output);
    levelOrderBottomHelper(node.right, level + 1, output);
}
```
