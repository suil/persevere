# Key insights
* Uses memoization with Map to cache results and avoid redundant recursive calls
* Recursive backtracking explores both possible character matches from s1 and s2
* String key `${p1}-${p2}-${p3}` represents unique state for position tracking
* Early termination when length check fails prevents unnecessary computation

# Code
```javascript
var isInterleave = function(s1, s2, s3) {
    if (s1.length + s2.length !== s3.length) { return false; }
    return memoize(s1, 0, s2, 0, s3, 0, new Map());
};

function memoize(s1, p1, s2, p2, s3, p3, memo) {
    const key = `${p1}-${p2}-${p3}`;
    if (memo.has(key)) {
        return memo.get(key);
    }

    if (p1 === s1.length && p2 === s2.length && p3 === s3.length) {
        memo.set(key, true);
        return true;
    }
    if (p1 < s1.length && s1[p1] == s3[p3] && memoize(s1, p1 + 1, s2, p2, s3, p3 + 1, memo)) {
        memo.set(key, true);
        return true;
    }
    if (p2 < s2.length && s2[p2] == s3[p3] && memoize(s1, p1, s2, p2 + 1, s3, p3 + 1, memo)) {
        memo.set(key, true);
        return true;
    }

    memo.set(key, false);
    return false;
}
```

```javascript
function dp(s1, s2, s3) {
    if (s1.length + s2.length !== s3.length) { return false; }

    const dp = [...Array(s1.length + 1)].map(_ => Array(s2.length + 1));
    for (let i = 0; i <= s1.length; i++) {
        for (let j = 0; j <= s2.length; j++) {
            if (i === 0 && j === 0) {
                dp[i][j] = true;
            } else if (i == 0) {
                dp[i][j] = dp[i][j - 1] && s2.charAt(j - 1) == s3.charAt(i + j - 1);
            } else if (j == 0) {
                dp[i][j] = dp[i - 1][j] && s1.charAt(i - 1) == s3.charAt(i + j - 1);
            } else {
                dp[i][j] = (dp[i - 1][j] && s1.charAt(i - 1) == s3.charAt(i + j - 1)) || (dp[i][j - 1] && s2.charAt(j - 1) == s3.charAt(i + j - 1));
            }
        }
    }
    return dp[s1.length][s2.length];
}
```
