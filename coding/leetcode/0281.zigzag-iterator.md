# Zigzag Iterator
[281. Zigzag Iterator](https://leetcode.com/problems/zigzag-iterator/)

Given two vectors of integers v1 and v2, implement an iterator to return their elements alternately.

/**
 * @constructor
 * @param {Integer[]} v1
 * @param {Integer[]} v1
 */
var ZigzagIterator = function ZigzagIterator(v1, v2) {
    this.currentV = v1; // 'v1' or 'v2'
    this.v1 = v1;
    this.v2 = v2;
    this.p1 = 0;
    this.p2 = 0;
};


/**
 * @this ZigzagIterator
 * @returns {boolean}
 */
ZigzagIterator.prototype.hasNext = function hasNext() {
    return this.p1 < this.v1.length || this.p2 < this.v2.length;
};

/**
 * @this ZigzagIterator
 * @returns {integer}
 */
ZigzagIterator.prototype.next = function next() {
    if (this.p1 >= this.v1.length && this.p2 >= this.v2.length) {
        return [];
    }

    if (this.currentV === this.v1) {
        if (this.p1 >= this.v1.length) {
            return this.v2[this.p2++];
        }
        this.currentV = this.v2;
        return this.v1[this.p1++];
    } else {
        if (this.p2 >= this.v2.length) {
            console.log(this.p1)
            return this.v1[this.p1++];
        }
        this.currentV = this.v1;
        return this.v2[this.p2++];
    }
};

/**
 * Your ZigzagIterator will be called like this:
 * var i = new ZigzagIterator(v1, v2), a = [];
 * while (i.hasNext()) a.push(i.next());
*/
