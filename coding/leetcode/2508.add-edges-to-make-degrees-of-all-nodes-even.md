# Add Edges to Make Degrees of All Nodes Even
[2508. Add Edges to Make Degrees of All Nodes Even](https://leetcode.com/problems/add-edges-to-make-degrees-of-all-nodes-even/)

There is an undirected graph consisting of n nodes numbered from 1 to n. You are given the integer n and a 2D array edges where edges[i] = [ai, bi] indicates that there is an edge between nodes ai and bi. The graph can be disconnected.

/**
 * @param {number} n
 * @param {number[][]} edges
 * @return {boolean}
 */
var isPossible = function(n, edges) {
    const adj = new Map();
    const degrees = new Array(n + 1).fill(0);

    for (let i = 1; i <= n; i++) {
        adj.set(i, new Set());
    }

    for (const [u, v] of edges) {
        adj.get(u).add(v);
        adj.get(v).add(u);
        degrees[u]++;
        degrees[v]++;
    }

    const oddNodes = [];
    for (let i = 1; i <= n; i++) {
        if (degrees[i] % 2 !== 0) {
            oddNodes.push(i);
        }
    }

    const numOdd = oddNodes.length;

    if (numOdd === 0) {
        return true;
    }

    // Case 1: Two odd-degree nodes
    else if (numOdd === 2) {
        const [a, b] = oddNodes;

        // Option 1: Add one edge (a, b)
        // Check if a and b are not already connected
        if (!adj.get(a).has(b)) {
            return true;
        }

        // Option 2: Add two edges involving a third node 'c'
        // Iterate through all possible nodes 'i' (1 to n)
        for (let i = 1; i <= n; i++) {
            // We need a node 'i' that is not connected to 'a' AND not connected to 'b'
            // And 'i' should not be 'a' or 'b' itself (implicitly handled by the `has` check)
            if (i !== a && i !== b && !adj.get(a).has(i) && !adj.get(b).has(i)) {
                return true;
            }
        }
        return false;
    }

    // Case 2: Four odd-degree nodes
    else if (numOdd === 4) {
        const [a, b, c, d] = oddNodes;

        // Try all three possible pairings to add two edges
        // Pairing 1: (a, b) and (c, d)
        if (!adj.get(a).has(b) && !adj.get(c).has(d)) {
            return true;
        }

        // Pairing 2: (a, c) and (b, d)
        if (!adj.get(a).has(c) && !adj.get(b).has(d)) {
            return true;
        }

        // Pairing 3: (a, d) and (b, c)
        if (!adj.get(a).has(d) && !adj.get(b).has(c)) {
            return true;
        }

        return false;
    }

    // Case 3: Any other number of odd-degree nodes (e.g., 1, 3, or > 4)
    else {
        return false;
    }
};
