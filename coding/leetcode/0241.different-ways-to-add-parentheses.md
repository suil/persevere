# Different Ways to Add Parentheses
[241. Different Ways to Add Parentheses](https://leetcode.com/problems/different-ways-to-add-parentheses/)

Given a string expression of numbers and operators, return all possible results from computing all the different possible ways to group numbers and operators. You may return the answer in any order.

# Code
```javascript
/**
 * @param {string} expression
 * @return {number[]}
 */
var diffWaysToCompute = function(expression) {
    if (expression.length === 0) return [];

    if (/^\d+$/.test(expression)) {
        return [Number(expression)];
    }
    
    const ways = [];
    for (let i = 0; i < expression.length; i++) {
        const op = expression[i];
        if (op !== '+' && op !== '*' && op !== '-') {
            continue;
        }
        
        const firstExp = expression.substring(0, i);
        const secondExp = expression.substring(i + 1);
        const firstRes = diffWaysToCompute(firstExp);
        const secondRes = diffWaysToCompute(secondExp);

        for (const first of firstRes) {
            for (const second of secondRes) {
                switch (op) {
                    case '+':
                        ways.push(first + second);
                        break
                    case '-':
                        ways.push(first - second);
                        break
                    case '*':
                        ways.push(first * second);
                        break
                }
            }
        }
    }
    return ways;
};
```
