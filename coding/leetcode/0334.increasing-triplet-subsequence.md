# Increasing Triplet Subsequence
[334. Increasing Triplet Subsequence](https://leetcode.com/problems/increasing-triplet-subsequence/)

Given an integer array nums, return true if there exists a triple of indices (i, j, k) such that i < j < k and nums[i] < nums[j] < nums[k]. If no such indices exists, return false.

# Code
```javascript
/**
 * @param {number[]} nums
 * @return {boolean}
 */
var increasingTriplet = function(nums) {
    if (nums.length < 3) return false;

    const letfMins = Array(nums.length).fill().map((_, index) => nums[index]);
    const rightMaxs = Array(nums.length).fill().map((_, index) => nums[index]);

    for (let i = 1; i < nums.length; i++) {
        letfMins[i] = Math.min(letfMins[i - 1], nums[i]);
    }

    for (let i = nums.length - 2; i >= 0; i--) {
        rightMaxs[i] = Math.max(rightMaxs[i + 1], nums[i]);
    }
    
    for (let i = 1; i < nums.length - 1; i++) {
        if (letfMins[i] < nums[i] && nums[i] < rightMaxs[i]) {
            return true;
        }
    }
    return false;
};
```
