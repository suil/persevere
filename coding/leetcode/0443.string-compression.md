/**
 * @param {character[]} chars
 * @return {number}
 */
var compress = function(chars) {
    const len = chars.length;
    let k = 0;
    let i = 0;
    while (i < len) {
        let j = i + 1;
        while (j < chars.length && chars[i] === chars[j]) {
            j++;
        }
        count = j - i;
        // if (j === chars.length) count += 1;
        console.log({ char: chars[i], i, j, count})
        chars[k++] = chars[i];
        if (count > 1) {
            const countStr = `${count}`;
            for (const c of countStr) {
                chars[k++] = c;
            }
        }
        i = j;
    }
    return k;
};
