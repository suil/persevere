# Longest Path With Different Adjacent Characters
[2246. Longest Path With Different Adjacent Characters](https://leetcode.com/problems/longest-path-with-different-adjacent-characters/)

You are given a tree (i.e. a connected, undirected graph that has no cycles) rooted at node 0 consisting of n nodes numbered from 0 to n - 1. The tree is represented by a 0-indexed array parent of size n, where parent[i] is the parent of node i. Since node 0 is the root, parent[0] == -1.

# Code
```javascript
/**
 * @param {number[]} parent
 * @param {string} s
 * @return {number}
 */
var longestPath = function(parent, s) {
    const n = parent.length;
    const tree = Array.from({ length: n }, () => []);
    
    for (let i = 1; i < n; i++) {
        tree[parent[i]].push(i);
    }

    let maxLength = -Infinity;

    const dfs = (node) => {
        if (tree[node].length === 0) return 1;

        let longest = 0;
        let secondLongest = 0;

        for (let child of tree[node]) {
            const childLength = dfs(child);

            if (s[child] !== s[node]) {
                if (childLength > longest) {
                    secondLongest = longest;
                    longest = childLength;
                } else if (childLength > secondLongest) {
                    secondLongest = childLength;
                }
            }
        }

        // Update max path length (1 for current node + two longest branches)
        maxLength = Math.max(maxLength, 1 + longest + secondLongest);

        // Return longest path from this node to leaf
        return 1 + longest;
    }

    dfs(0);
    return maxLength === -Infinity ? 1 : maxLength;
};
```
