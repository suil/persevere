```javascript
/**
 * @param {string} word
 * @return {string}
 */
var compressedString = function(word) {
    let i = 0;
    let res = '';

    while (i < word.length) {
        let count = 1;
        while (i + 1 < word.length && word[i + 1] === word[i] && count < 9) {
            i++;
            count++;
        }
        res += `${count}${word[i]}`;
        i++;
    }

    return res;
};
```
