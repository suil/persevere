# Key insights
* Uses BFS to find minimum steps by exploring all possible square subtractions level by level
* Queue stores remainders after subtracting perfect squares from current numbers
* Level-by-level processing ensures first time remainder=0 gives optimal solution
* Explores all perfect squares â‰¤ current number to generate next level candidates

```javascript
var numSquares = function(n) {
    let queue = [n];
    let count = 0;

    while (queue.length > 0) {
        const nextQueue = [];
        count++;
        
        for (const item of queue) {
            for (let i = 1; i * i <= item; i++) {
                const remainder = item - i * i;
                if (remainder === 0) {
                    return count;
                }

                nextQueue.push(remainder);
            }
            
        }

        queue = nextQueue;
    }

    return -1;
};
```

```javascript
var numSquares = function(n) {
    const dp = [...Array(n + 1)].map((_, index) => index);

    for (let i = 1; i <= n; i++) {
        for (let j = 1; j * j <= i; j++) {
            dp[i] = Math.min(dp[i], 1 + dp[i - j * j]);
        }
    }

    return dp[n];
};
```
