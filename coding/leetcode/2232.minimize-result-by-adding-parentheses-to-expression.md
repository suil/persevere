# Minimize Result by Adding Parentheses to Expression
[2232. Minimize Result by Adding Parentheses to Expression](https://leetcode.com/problems/minimize-result-by-adding-parentheses-to-expression/)

You are given a 0-indexed string expression of the form "<num1>+<num2>" where <num1> and <num2> represent positive integers.

# Code
```javascript
/**
 * @param {string} expression
 * @return {string}
 */
var minimizeResult = function(expression) {
    const plusIndex = expression.indexOf('+');
    const leftStr = expression.slice(0, plusIndex);
    const rightStr = expression.slice(plusIndex + 1);

    let min = Infinity;
    let res = '';

    for (let left = 0; left < leftStr.length; left++) {
        for (let right = 1; right <= rightStr.length; right++) { // need to be [1, rightStr.length]
            let left1 = leftStr.slice(0, left);
            let left2 = leftStr.slice(left);
            let right1 = rightStr.slice(0, right);
            let right2 = rightStr.slice(right);

            let part1 = left1.length > 0 ? Number(left1) : 1;
            let part2 = Number(left2);
            let part3 = Number(right1);
            let part4 = right2.length > 0 ? Number(right2) : 1;

            const val = part1 * (part2 + part3) * part4;

            if (val < min) {
                min = val;
                res = `${left1}(${left2}+${right1})${right2}`;
            }
        }
    }
    return res;
};
```
