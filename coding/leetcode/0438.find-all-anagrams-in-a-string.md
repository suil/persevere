# Find All Anagrams in a String
[438. Find All Anagrams in a String](https://leetcode.com/problems/find-all-anagrams-in-a-string/)

Given two strings s and p, return an array of all the start indices of p's anagrams in s. You may return the answer in any order.

# Code

```javascript
/**
 * @param {string} s
 * @param {string} p
 * @return {number[]}
 */
var findAnagrams = function(s, p) {
    function isEqual(mapS, mapP) {
        for (const [key, count] of mapP) {
            if (mapS.get(key) != count) return false;
        }

        return true;
    }

    const countS = new Map();
    const countP = new Map();

    for (let i = 0; i < p.length; i++) {
        countP.set(p[i], (countP.get(p[i]) || 0) + 1);
        countS.set(s[i], (countS.get(s[i]) || 0) + 1);
    }

    const res = [];
    if (isEqual(countS, countP)) res.push(0);

    for (let i = p.length; i < s.length; i++) {
        countS.set(s[i], (countS.get(s[i]) || 0) + 1);
        countS.set(s[i - p.length], countS.get(s[i - p.length]) - 1);
        if (isEqual(countS, countP)) {
            res.push(i - p.length + 1);
        }
    }

    return res;
};
```
