# Trapping Rain Water
[42. Trapping Rain Water](https://leetcode.com/problems/trapping-rain-water/)

# Key insights
* Pre-compute maximum heights from left and right to avoid redundant calculations
* Water level at each position is limited by the minimum of left and right maximums
* Three-pass approach: left max array, right max array, then calculate trapped water
* Dynamic programming technique using auxiliary arrays for O(n) time complexity

# Code
```javascript
var trap = function(height) {
    let left = 0;
    let right = height.length - 1;
    let water = 0;
    let leftMax = 0;
    let rightMax = 0;

    while (left < right) {
        if (height[left] < height[right]) {
            leftMax = Math.max(leftMax, height[left]);
            water += leftMax - height[left];
            left++;
        } else {
            rightMax = Math.max(rightMax, height[right]);
            water += rightMax - height[right];
            right--;
        }
    }
    return water
};
```

/**
 * @param {number[]} height
 * @return {number}

var trap = function(height) {
    const stack = [];
    let sum = 0;

    height = [0, ...height, 0]
    for (let i = 0; i < height.length; i++) {
        const curr = height[i];
        while (stack.length > 0 && height[stack.at(-1)] <= curr) {
            const index = stack.pop();
            if (stack.length > 0) {
                const nextLargerIndex = i;
                const prevLargerOrEqualIndex = stack.at(-1);
                const h = Math.min(
                    height[nextLargerIndex],
                    height[prevLargerOrEqualIndex],
                ) - height[index];
                const w = nextLargerIndex - prevLargerOrEqualIndex - 1;
                sum += h * w;
            }
        }
        stack.push(i);
    }
    return sum;
};

```javascript
var trap = function(height) {
    if (!height) { return 0; }
    
    let len = height.length;
    let leftMax = [], rightMax = [];
    
    leftMax[0] = height[0];
    rightMax[len - 1] = height[len - 1];
    
    for (let i = 1; i < len; i++) {
        leftMax[i] = Math.max(height[i], leftMax[i - 1])
    }
    for (let i = len - 2; i >= 0; i--) {
        rightMax[i] = Math.max(height[i], rightMax[i + 1]);
    }
    
    let totalWater = 0;
    for (let i = 0; i < len; i++) {
        let water = Math.min(leftMax[i], rightMax[i]) - height[i];
        totalWater += water;
    }
    return totalWater;
};
```

# Complexity
**Solution 1:**
**Time:** O(n) - Three separate loops each iterate through the array once
**Space:** O(n) - Two additional arrays (leftMax and rightMax) store maximum values for each position

**Solution 2:**
**Time:** O(n) - Single while loop processes each element exactly once using two pointers
**Space:** O(1) - Only uses a constant amount of extra variables regardless of input size
