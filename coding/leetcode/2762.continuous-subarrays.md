/**
 * @param {number[]} nums
 * @return {number}
 */
var continuousSubarrays = function(nums) {
    let count = 0;
    let left = 0;
    const minQ = [];
    const maxQ = [];

    for (let right = 0; right < nums.length; right++) {
        const rightNum = nums[right];

        while (minQ.length > 0 && nums[minQ.at(-1)] >= rightNum) minQ.pop();
        minQ.push(right);
        
        while (maxQ.length > 0 && nums[maxQ.at(-1)] <= rightNum) maxQ.pop();
        maxQ.push(right);

        while (left <= right && nums[maxQ[0]] - nums[minQ[0]] > 2) {
            if (minQ[0] === left) minQ.shift();
            if (maxQ[0] === left) maxQ.shift();
            left++;
        }

        count += right - left + 1;
    }

    return count;
};
