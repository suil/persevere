# Count Subarrays With Score Less Than K
[2302. Count Subarrays With Score Less Than K](https://leetcode.com/problems/count-subarrays-with-score-less-than-k/)

The score of an array is defined as the product of its sum and its length.

```javascript
/**
 * @param {number[]} nums
 * @param {number} k
 * @return {number}
 */
var countSubarrays = function(nums, k) {
    const n = nums.length;
    const prefixSum = new Array(n + 1).fill(0);
    for (let i = 0; i < n; i++) {
        prefixSum[i + 1] = prefixSum[i] + nums[i];
    }

    let totalCount = 0;

    for (let right = 1; right <= n; right++) {
        let low = 0;
        let high = right;
        let maxValidLength = 0;

        while (low <= high) {
            const mid = Math.floor(low + (high - low) / 2);

            // if (mid === 0) {
            //     low = mid + 1;
            //     continue;
            // }

            const currentSubarraySum = prefixSum[right] - prefixSum[right - mid];
            const currentScore = currentSubarraySum * mid;

            if (currentScore < k) {
                maxValidLength = mid;
                low = mid + 1;
            } else {
                high = mid - 1;
            }
        }
        totalCount += maxValidLength;
    }

    return totalCount;    

};
```
