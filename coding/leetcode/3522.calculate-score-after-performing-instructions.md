You are given two arrays, instructions and values, both of size n.

# Code
```javascript
/**
 * @param {string[]} instructions
 * @param {number[]} values
 * @return {number}
 */
var calculateScore = function(instructions, values) {
    const n = instructions.length;
    const visited = new Set();

    function backtrack(index) {
        if (index < 0 || index >= n) return 0;
        if (visited.has(index)) return 0;

        visited.add(index);

        let score = 0;
        if (instructions[index] === 'add') {
            score = values[index] + backtrack(index + 1);
        } else if (instructions[index] === 'jump') {
            score = backtrack(index + values[index]);
        }

        visited.delete(index);
        return score;
    }

    return backtrack(0);
};
```
