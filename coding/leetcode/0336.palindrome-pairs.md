# Palindrome Pairs
[336. Palindrome Pairs](https://leetcode.com/problems/palindrome-pairs/)

You are given a 0-indexed array of unique strings words.

# Key insights
- Store all words in a map for O(1) lookup by word -> index
- For each word, try all prefix/suffix splits
- If prefix is palindrome, look for reversed suffix as another word (prepend it)
- If suffix is palindrome, look for reversed prefix as another word (append it)
- Handle edge cases: empty strings and avoiding self-pairing

# Code
```javascript
/**
 * @param {string[]} words
 * @return {number[][]}
 */
var palindromePairs = function(words) {
    const map = new Map();
    for (let i = 0; i < words.length; i++) {
        map.set(words[i], i);
    }

    function isPalindrome(str) {
        if (str.length === 0) return true;
        let left = 0, right = str.length - 1;
        while (left < right) {
            if (str[left++] !== str[right--]) return false;
        }
        return true;
    }

    function reverse(str) {
        return str.split('').reverse().join('');
    }

    const res = [];
    for (let wordIndex = 0; wordIndex < words.length; wordIndex++) {
        const word = words[wordIndex];
        for (let i = 0; i <= word.length; i++) {
            const prefix = word.slice(0, i);
            const suffix = word.slice(i);
            
            if (isPalindrome(prefix)) {
                const reversed = reverse(suffix);
                if (map.has(reversed) && map.get(reversed) !== wordIndex) {
                   res.push([map.get(reversed), wordIndex]) ;
                }
            }

            if (i !== word.length && isPalindrome(suffix)) {
                const reversed = reverse(prefix);
                if (map.has(reversed) && map.get(reversed) !== wordIndex) {
                    res.push([wordIndex, map.get(reversed)]);
                }
            }
        }
    }

    return res;
};
```

# Complexity
**Time:** O(n * k^2) - where n is number of words and k is max word length
**Space:** O(n * k) - for storing words in the map
