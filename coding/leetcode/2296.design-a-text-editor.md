/**
 * @class TextEditor
 * Uses two stacks (arrays) to represent characters left/right of the cursor.
 */
var TextEditor = function() {
    this.left = [];   // characters to the left of the cursor (end is immediate left)
    this.right = [];  // characters to the right of the cursor (end is immediate right)
};

/**
 * @param {string} text
 * @return {void}
 */
TextEditor.prototype.addText = function(text) {
    for (let ch of text) {
        this.left.push(ch);
    }
};

/**
 * @param {number} k
 * @return {number} number of characters actually deleted
 */
TextEditor.prototype.deleteText = function(k) {
    let deleted = 0;
    while (deleted < k && this.left.length > 0) {
        this.left.pop();
        deleted++;
    }
    return deleted;
};

/**
 * @param {number} k
 * @return {string} up to 10 characters to the left of the cursor
 */
TextEditor.prototype.cursorLeft = function(k) {
    let moved = 0;
    while (moved < k && this.left.length > 0) {
        this.right.push(this.left.pop());
        moved++;
    }
    return this._getLast10();
};

/**
 * @param {number} k
 * @return {string} up to 10 characters to the left of the cursor
 */
TextEditor.prototype.cursorRight = function(k) {
    let moved = 0;
    while (moved < k && this.right.length > 0) {
        this.left.push(this.right.pop());
        moved++;
    }
    return this._getLast10();
};

/**
 * Helper: returns up to 10 chars to the left of cursor as a string
 */
TextEditor.prototype._getLast10 = function() {
    const start = Math.max(0, this.left.length - 10);
    return this.left.slice(start).join('');
};

/** 
 * Your TextEditor object will be instantiated and called as such:
 * var obj = new TextEditor()
 * obj.addText(text)
 * var param_2 = obj.deleteText(k)
 * var param_3 = obj.cursorLeft(k)
 * var param_4 = obj.cursorRight(k)
 */
