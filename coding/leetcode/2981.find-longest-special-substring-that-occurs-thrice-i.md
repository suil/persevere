/**
 * @param {string} s
 * @return {number}
 */
var maximumLength = function(s) {
    const count = new Map();

    for (let right = 0; right < s.length; right++) {
        let currString = "";
        for (let end = right; end < s.length; end++) {
            if (currString.length === 0 || currString[currString.length - 1] === s[end]) {
                currString += s[end];
                count.set(currString, (count.get(currString) || 0) + 1);
            } else {
                break;
            }
        }
    }

    let ans = 0;
    for (const str of count.keys()) {
        if (count.get(str) >= 3 && str.length > ans) {
            ans = str.length;
        }
    }

    return ans === 0 ? -1 : ans;
};
