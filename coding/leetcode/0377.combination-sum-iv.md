# Key insights
* Top-down dynamic programming with memoization to avoid recomputing subproblems
* Base case: target=0 returns 1 (one way to make zero with empty combination)
* For each target, try subtracting every number and sum all valid recursive results
* HashMap stores results for each target value, reducing time complexity from exponential to O(target*nums.length)

var combinationSum4 = function(nums, target) {
    const memo = new Map()
    return memoize(nums, target, memo);
};

function memoize(nums, target, memo) {
    if (target === 0) {
        return 1;
    }

    if (memo.has(target)) {
        return memo.get(target);
    }

    let res = 0;
    for (const num of nums) {
        if (target - num >= 0) {
            res += memoize(nums, target - num, memo);
        }
    }
    memo.set(target, res);
    return res;
}
