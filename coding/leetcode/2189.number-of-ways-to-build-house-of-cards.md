/**
 * @param {number} n
 * @return {number}
 */
var houseOfCards = function(n) {
    const memo = new Map();

    const solve = (remCards, prevT) => {
        const key = `${remCards},${prevT}`;
        if (memo.has(key)) {
            return memo.get(key);
        }

        let totalCount = 0;
        const maxT = Math.min(prevT - 1, Math.floor((remCards + 1) / 3));

        for (let t = 1; t <= maxT; t++) {
            const cardsUsed = 3 * t - 1;
            const newRemCards = remCards - cardsUsed;

            if (newRemCards === 0) {
                totalCount++;
            } else if (newRemCards > 0) {
                totalCount += solve(newRemCards, t);
            }
        }

        memo.set(key, totalCount);
        return totalCount;
    };

    let totalDistinctHouses = 0;
    
    if ((n + 1) % 3 === 0) {
        totalDistinctHouses = 1;
    }

    const maxT = Math.floor((n + 1) / 3);
    for (let t = 1; t <= maxT; t++) {
        const cardsUsed = 3 * t - 1;
        const remCards = n - cardsUsed;
        
        if (remCards > 0) {
            totalDistinctHouses += solve(remCards, t);
        }
    }

    return totalDistinctHouses;
};
