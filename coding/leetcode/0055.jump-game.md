# Key insights
* Uses memoization to avoid recomputing same subproblems in recursive solution
* Dynamic programming with top-down approach exploring all possible jumps from each position
* HashMap stores results for each index to achieve optimal substructure
* Recursively tries all jump lengths from current position until target is reached

# Code
```javascript
function memoize(nums, currentIndex, memo) {
    if (memo.has(currentIndex)) {
        return memo.get(currentIndex);
    }
    if (currentIndex > nums.length) {
        return false;
    }
    if (currentIndex === nums.length - 1) {
        return true;
    }

    const maxNumSteps = nums[currentIndex];
    for (let i = 1; i <= maxNumSteps; i++) {
        if (memoize(nums, currentIndex + i, memo)) {
            memo.set(currentIndex, true);
            return true;
        }
    }
    memo.set(currentIndex, false);
    return false;
}
```

```javascript
function dp(nums) {
    const dp = Array(nums.length).fill(false);
    dp[nums.length - 1] = true;

    for (let i = nums.length - 2; i >= 0; i--) {
        for (let j = 1; j <= nums[i]; j++) {
            dp[i] = dp[i] || (dp[i + j] || false);
        }
    }
    return dp[0];
}
```
