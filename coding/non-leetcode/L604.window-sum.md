```javascript
  winSum(nums, k) {
    // write your code here
    if (nums == null || nums.length < k || k <= 0) {
        return [];
    }

    const sums = Array(nums.length - k + 1).fill(0);
    for (let i = 0; i < k; i++) {
        sums[0] += nums[i];
    }

    for (let i = 1; i < sums.length; i++) {
        sums[i] = sums[i - 1] - nums[i - 1] + nums[i + k - 1];
    }

    return sums;
  }
```
